#include<stdio.h>#include<stdlib.h>struct queue{int data;struct queue * link;};struct queue * front=NULL;struct queue * rear;void insert(int d){struct queue *node;node=(struct queue *)malloc(sizeof(struct queue));node->data=d;printf("\n %d is inserted at the rear ",node->data);node->link=NULL;if(front==NULL){front=node;rear=node;}else{rear->link=node;rear=node;}}void delete(void){struct queue * temp;if(front==NULL){printf("\n UNDERFLOW....... ");}else{printf("\n %d is deleted from the front ",front->data);temp=front->link;free(front);front=temp;}}void print(void){struct queue* temp;temp =front;printf("\n");while(temp!= NULL){printf(" %d --->",temp->data);temp=temp->link;}printf(" NULL ");}void search(int s){if(front==NULL)printf("\n UNDERFLOW....... ");else{struct queue * temp;temp=front;int y=1;while(temp!=NULL){if(temp->data==s){printf("\n %d is found in the queue ",temp->data);y=0;}temp=temp->link;}if(y)printf("\n %d is not found in the queue ",s);}}main(){insert(1);insert(2);delete();insert(3);print();delete();delete();delete();print();insert(22);insert(6);search(22);search(6);search(3);}
